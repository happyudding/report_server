table : products - 제품 기본 정보
columns
컬럼명 타입 제약
id INTEGER - PK, AUTOINCREMENT # 내부 ID
part_id TEXT - NOT NULL, UNIQUE # 부품 번호
product_group TEXT # 제품군
osat TEXT # 외부 업체
prod_type TEXT # 생산 유형 (MP, ER)
dev_manager TEXT # 개발 담당자
mass_prod_manager TEXT # 양산 담당자
pe_manager TEXT # PE 담당자
modify_date TEXT # 최종 수정일시 (YYYY-MM-DD HH:MM:SS)
status TEXT # 상태 (완결 / 상신취소 / 반려)
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP # 레코드 생성일시

table : product_details - 제품 상세 메타
id INTEGER PK, AUTOINCREMENT # 내부 ID
product_id INTEGER NOT NULL, FK ->products.id # 부모 제품 참조
title TEXT - #상세 타이블 (현재 전부 "ONE")
fetched_at TIMESTAMP # 데이터 수집 일시 (ISO 8601)

table : product_detail_tables - 공정 단계별 파라미터
id INTEGER PK, AUTOINCREMENT # 내부 ID
detail_id INTEGER NOT NULL, FK -> product_details.id 부모 상세 참조
table_index INTEGER DEFAULT 0 # 상세 내 순번
osat TEXT # 공정 담당 osat
step TEXT # 공정 단계 L1 L2 L3 FT M1~M4
temperature TEXT # 온도 설정값
duration TEXT # 공정 시간
abs_distance_y TEXT # 절대거리 Y
abs_distance_x TEXT # 절대거리 X
sanding TEXT # CHIP or S/W
sanding_value TEXT # 값
equip TEXT # 설비 유형 T2K ETS UFLEX ND4 IFLEX V93K
equip_name TEXT # 설비 모델명
para TEXT # 파라미터값 1~1024 (보통 6 12 16 24 48 128 512 등)
flat_zone TEXT # 0 or 270 or 90 or 180 등
pgm_name TEXT # 프로그램 이름 
pgm_dir TEXT # 프로그램 디렉토리
